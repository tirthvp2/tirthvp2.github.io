std::vector filterMuons(std::vector& data) {
    std::vector filtered;
    for (double value : data) {
        if (value > 1000) { // Threshold for muon detection
            filtered.push_back(value);
        }
    }
    return filtered;
}

int main() {
    std::vector data = {500, 1200, 800, 1500};
    auto muons = filterMuons(data);
    for (double muon : muons) {
        std::cout << "Detected Muon: " << muon << std::endl;
    }
    return 0;
}

import com.google.gson.Gson;
import java.net.HttpURLConnection;
import java.net.URL;
import java.util.Scanner;

public class TriviaApi {
    public static String fetchTrivia() {
        try {
            URL url = new URL("https://api.trivia.com/questions");
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");
            Scanner scanner = new Scanner(conn.getInputStream());
            StringBuilder response = new StringBuilder();
            while (scanner.hasNext()) {
                response.append(scanner.nextLine());
            }
            scanner.close();
            return response.toString();
        } catch (Exception e) {
            e.printStackTrace();
            return null;
        }
    }

    public static void main(String[] args) {
        String json = fetchTrivia();
        Gson gson = new Gson();
        TriviaQuestion[] questions = gson.fromJson(json, TriviaQuestion[].class);
        System.out.println(questions[0].getQuestion());
    }
}
