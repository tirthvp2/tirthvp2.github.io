Python Code Demo

A live Python script (via Pyodide) generating a Sierpinski Triangle fractal pattern.

def sierpinski(n):
    if n == 0:
        return ['*']
    smaller = sierpinski(n - 1)
    width = len(smaller[0])
    space = ' ' * (width // 2)
    result = []
    for line in smaller:
        result.append(space + line + space)
    for line in smaller:
        result.append(line + ' ' + line)
    return result

# Generate Sierpinski Triangle with depth 3
triangle = sierpinski(3)
'\n'.join(triangle)
